Table Product {
  "id" int [pk, not null, increment]
  "name" text
  "preview_path" text
  "price" int
  "description" text
  "quantity" int

  "rate" int
  "created_at" timestamp(0)
  "deleted_at" timestamp(0)
}

Table Customer {
  "id" int [pk, not null, increment]
  "name" text
  "password" text
  "avatar_url" text
  "phone" text
  "address" text
  "email" text
  "updated_at" timestamp(0)
  "created_at" timestamp(0)
  "deleted_at" timestamp(0)
}

Table Cart {
  "id" int [pk, not null, increment]
  "productId" int [not null, ref: > Product.id]
  "customerId" int [not null, ref: > Customer.id]
  "quantity" int
  "total_money" int
  "updated_at" timestamp(0)
  "created_at" timestamp(0)
  "deleted_at" timestamp(0)
}

Table Blog {
  "id" int [pk, not null, increment]
  "title" text
  "description" text
  "preview_path" text
  "updated_at" timestamp(0)
  "created_at" timestamp(0)
  "deleted_at" timestamp(0)
}

Table Admin {
  "id" int [pk, not null, increment]
  "name" text
  "password" text
  "avatar_url" text
  "updated_at" timestamp(0)
  "created_at" timestamp(0)
  "deleted_at" timestamp(0)
}
